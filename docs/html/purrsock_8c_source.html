14const char *ps_result_to_cstr(ps_result_t result) {
15 switch (result) {
16 case PS_CONNCLOSED: return "Connection closed";
17 case PS_ERROR_INTERNAL: return "Internal error";
18 case PS_SUCCESS: return "Success";
19 case PS_ERROR_NOTINIT: return "Error not initialized";
20 case PS_ERROR_MSGTOOLONG: return "Error msg too long";
21 case PS_ERROR_ADDRINUSE: return "Error addr in use";
22 case PS_ERROR_ADDRNOTAVAIL: return "Error addr not avail";
23 case PS_ERROR_NETDOWN: return "Error net down";
24 case PS_ERROR_NETRESET: return "Error net reset";
25 case PS_ERROR_CONNRESET: return "Error conn reset";
26 case PS_ERROR_CONNREFUSED: return "Error conn refused";
27 case PS_ERROR_HOSTDOWN: return "Error host down";
28 case PS_ERROR_SHUTDOWN: return "Error shut down";
29 case PS_ERROR_UNKNOWN: return "Unknown error";
30 }
31 assert(0 && "Unreachable");
32 return NULL;
33}
