41ps_result_t _last_ps_result(const char *func_name) {
42 int error = WSAGetLastError();
43 ps_result_t result = PS_ERROR_UNKNOWN;
44
45 switch (error) {
46 case 0: result = PS_SUCCESS; break;
47 case WSANOTINITIALISED: result = PS_ERROR_NOTINIT; break;
48 case WSAEMSGSIZE: result = PS_ERROR_MSGTOOLONG; break;
49 case WSAEADDRINUSE: result = PS_ERROR_ADDRINUSE; break;
50 case WSAEADDRNOTAVAIL: result = PS_ERROR_ADDRNOTAVAIL; break;
51 case WSAENETDOWN: result = PS_ERROR_NETDOWN; break;
52 case WSAENETRESET: result = PS_ERROR_NETRESET; break;
53 case WSAECONNRESET: result = PS_ERROR_CONNRESET; break;
54 case WSAECONNREFUSED: result = PS_ERROR_CONNREFUSED; break;
55 case WSAEHOSTDOWN: result = PS_ERROR_HOSTDOWN; break;
56 case WSAESHUTDOWN: result = PS_ERROR_SHUTDOWN; break;
57 case WSAETIMEDOUT: result = PS_ERROR_TIMEOUT; break;
58
59 default:
60 result = PS_ERROR_UNKNOWN;
61 log_error("Error in function %s: Unknown error. WSA Error Code: %d\n", func_name, error);
62 return result;
63 }
64
65 log_error("Error in function %s: %s\n", func_name, ps_result_to_cstr(result));
66
67 return result;
68}

73 SOCKET socket;
74 struct sockaddr_in addr;
75} _purrsock_socket_data_t;

77ps_result_t _purrsock_create_socket(_purrsock_socket_t *in_socket) {
78 assert(in_socket);
79
80 _purrsock_socket_data_t *data = (_purrsock_socket_data_t*)malloc(sizeof(*data));
81 assert(data);
82 memset(data, 0, sizeof(*data));
83
84 ps_result_t result = PS_SUCCESS;
85 switch (in_socket->protocol) {
86 case PS_PROTOCOL_TCP: {
87 data->socket = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP);
88 } break;
89 case PS_PROTOCOL_UDP: {
90 data->socket = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
91 } break;
92 default: {
93 assert(0 && "Unreachable");
94 result = PS_ERROR_INTERNAL;
95 }
96 }
97
98 if (data->socket == INVALID_SOCKET)
99 result = _last_ps_result("purrsock_create_socket");
100
101 in_socket->data = data;
102
103 if (result) { // Error
104 _purrsock_destroy_socket(in_socket);
105 free(data);
106 }
107
108 return result;
109}

134 assert(socket);
135
136 _purrsock_socket_data_t *data = (_purrsock_socket_data_t*)socket->data;
137 if (!data) return PS_ERROR_NOTINIT;
138
139 struct sockaddr_in addr = {0};
140 addr.sin_family = AF_INET;
141 addr.sin_addr.s_addr = ip?inet_addr(ip):INADDR_ANY;
142 addr.sin_port = htons(port);
143
144 if (bind(data->socket, (struct sockaddr*)&addr, sizeof(addr)) == SOCKET_ERROR) return _last_ps_result("purrsock_create_socket_from_addr");

159ps_result_t _purrsock_accept_socket(_purrsock_socket_t *socket, _purrsock_socket_t **client) {
160 assert(socket && client);
161 _purrsock_socket_data_t *data = (_purrsock_socket_data_t*)socket->data;
162 if (!data) return PS_ERROR_NOTINIT;
163
164 struct sockaddr_in addr = {0};
165 int addr_size = sizeof(addr);
166 SOCKET sock = accept(data->socket, (struct sockaddr*)&addr, &addr_size);
167 if (sock == INVALID_SOCKET) return _last_ps_result("purrsock_accept_socket");
168
169 *client = (_purrsock_socket_t*)malloc(sizeof(**client));
170 assert(*client);
171 (*client)->protocol = socket->protocol;
172
173 _purrsock_socket_data_t *client_data = (_purrsock_socket_data_t*)malloc(sizeof(*client_data));
174 assert(client_data);
175 client_data->socket = sock;
176 (*client)->data = client_data;
177
178 return PS_SUCCESS;
179}

197ps_result_t _purrsock_read_socket_packet(_purrsock_socket_t *socket, ps_packet_t *packet, _purrsock_socket_t **from) {
198 assert(socket && packet);
199 _purrsock_socket_data_t *data = (_purrsock_socket_data_t*)socket->data;
200 if (!data) return PS_ERROR_NOTINIT;
201
202 int res = 0;
203 switch (socket->protocol) {
204 case PS_PROTOCOL_TCP: {
205 res = recv(data->socket, packet->buf, packet->capacity, 0);
206 } break;
207 case PS_PROTOCOL_UDP: {
208 assert(from);
209 _purrsock_socket_data_t *from_data = (_purrsock_socket_data_t*)malloc(sizeof(*from_data));
210 int fromlen = sizeof(struct sockaddr_in);
211 res = recvfrom(data->socket, packet->buf, packet->capacity, 0, (struct sockaddr*)&from_data->addr, &fromlen);
212 if (res > 0) {
213 *from = (_purrsock_socket_t*)malloc(sizeof(**from));
214 assert(*from);
215 (*from)->protocol = socket->protocol;
216 (*from)->data = from_data;
217 }
218 } break;
219 default: {
220 assert(0 && "Unreachable");
221 return PS_ERROR_INTERNAL;
222 }
223 }
224
225 if (res < 0) return _last_ps_result("purrsock_read_socket_packet");
226 else if (res == 0) return PS_CONNCLOSED;
227 packet->size = (size_t)res;
228 return PS_SUCCESS;
229}

231ps_result_t _purrsock_send_socket_packet(_purrsock_socket_t *socket, ps_packet_t packet, _purrsock_socket_t *to) {
232 assert(socket && packet.buf);
233 _purrsock_socket_data_t *data = (_purrsock_socket_data_t*)socket->data;
234 if (!data) return PS_ERROR_NOTINIT;
235
236 int res = 0;
237 char *ptr = packet.buf;
238 while (packet.size > 0 && res >= 0) {
239 switch (socket->protocol) {
240 case PS_PROTOCOL_TCP: {
241 res = send(data->socket, ptr, packet.size, 0);
242 } break;
243 case PS_PROTOCOL_UDP: {
244 assert(to);
245 _purrsock_socket_data_t *to_data = (_purrsock_socket_data_t*)to->data;
246 int tolen = sizeof(to_data->addr);
247 res = sendto(data->socket, ptr, packet.size, 0, (struct sockaddr*)&to_data->addr, tolen);
248 } break;
249 default: {
250 assert(0 && "Unreachable");
251 return PS_ERROR_INTERNAL;
252 }
253 }
254
255 if (res < 0) break;
256 packet.size -= res;
257 ptr += res;
258 }
259
260 if (res == SOCKET_ERROR) return _last_ps_result("purrsock_send_socket_packet");
261 return PS_SUCCESS;
262}
